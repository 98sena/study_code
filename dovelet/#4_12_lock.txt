#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int num[4];
	int cnt = 0;
	int sum = 0;
	while (cnt <3) {
		cnt = 0; sum = 0;
		for (int i = 0; i < 4; i++) {
			cin >> num[i];
		}
		for (int i = 0; i <3; i++) {
			cout << i << "는 " << sum << endl;
			if (num[i] == 0) {
				cnt++;
			}
			if (cnt == 4) {
				return 0;
			}
			sum += (num[i + 1] - num[i]) * 9;
		}
		cout << sum + 360 * 3;
	}
	return 0;
}
