#include <iostream>
#include <algorithm>
#include <cmath>
using namespace std;

int main() {
	int n;
	int mul;
	double son[100];
	cin >> n >> mul;
	for (int i = 0; i < n; i++) {
		cin >> son[i];
		son[i] = pow(son[i], mul);
	}
	int sum = 0;
	for (int i = 0; i < n; i++) {
		if (son[i] > 0) {
			sum += son[i];
		}
	}
	cout << sum;
	return 0;
}
