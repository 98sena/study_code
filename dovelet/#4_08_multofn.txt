#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int n;
	cin >> n;
	int cnt = 0;
	for (int i = 1; i < 4; i++) {
		for (int j = 1; j < 4; j++) {
			for (int k = 1; k < 4; k++) {
				for (int l = 1; l < 4; l++) {
					if ((i * 1000 + j * 100 + k * 10 + l) % n == 0)  cnt++;
				}
			}
		}
	}
	cout << cnt;
	return 0;
} //0.28ÃÊ
